    6. Implement `#resolveRound()`:
    ```javascript
    #resolveRound() {
      const playerHand = this.#playerHands[this.#activeHandIndex];
      const dealerValue = this.#dealerHand.value;
      const playerValue = playerHand.value;
      let outcome, payout;

      if (this.#dealerHand.isBust) {
        outcome = 'WIN';
        payout = this.#currentBet * 2; // bet back + 1:1 winnings
      } else if (playerValue > dealerValue) {
        outcome = 'WIN';
        payout = this.#currentBet * 2;
      } else if (playerValue < dealerValue) {
        outcome = 'LOSS';
        payout = 0;
      } else {
        outcome = 'PUSH';
        payout = this.#currentBet; // bet returned
      }

      this.#chips += payout;
      this.#result = {
        outcome,
        playerValue,
        dealerValue,
        payout,
        handResults: [{ outcome, payout }]
      };
      this.#phase = 'ROUND_OVER';
    }
    ```
    NOTE: payout for WIN is `bet * 2` because the bet was already deducted from chips during placeBet. So chips += bet*2 means: original bet returned + 1:1 winnings.

    7. Implement `doubleDown()`:

    ```javascript
    doubleDown() {
      if (this.#phase !== 'PLAYER_TURN') throw new Error('Can only double down during PLAYER_TURN phase');
      const hand = this.#playerHands[this.#activeHandIndex];
      if (hand.cards.length !== 2) throw new Error('Can only double down on first two cards');
      if (this.#chips < this.#currentBet) throw new Error('Insufficient chips to double down');

      // Double the bet
      this.#chips -= this.#currentBet;
      this.#currentBet *= 2;
      this.#playerHands[this.#activeHandIndex] = { ...hand, bet: this.#currentBet };

      // Draw exactly one card
      const card = this.#deck.draw();
      this.#playerHands[this.#activeHandIndex] = addCardToHand(this.#playerHands[this.#activeHandIndex], card);

      if (this.#playerHands[this.#activeHandIndex].isBust) {
        this.#result = {
          outcome: 'LOSS',
          playerValue: this.#playerHands[this.#activeHandIndex].value,
          dealerValue: this.#dealerHand.value,
          payout: 0,
          handResults: [{ outcome: 'LOSS', payout: 0 }]
        };
        this.#revealDealerHoleCard();
        this.#phase = 'ROUND_OVER';
      } else {
        // Dealer plays
        this.#phase = 'DEALER_TURN';
        this.#playDealerTurn();
        this.#resolveRound();
      }
      return this.#getState();
    }
    ```
